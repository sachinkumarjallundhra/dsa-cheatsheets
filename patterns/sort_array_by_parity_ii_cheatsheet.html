The Sort Array By Parity II problem (LeetCode 922) asks you to rearrange an array so that even-indexed positions have even numbers and odd-indexed positions have odd numbers. The input array has equal numbers of even and odd integers.

ğŸ§± Pattern Template (Two Pointers Solution)

class Solution {
    public int[] sortArrayByParityII(int[] nums) {
        int n = nums.length;
        int even = 0, odd = 1;
        while (even < n && odd < n) {
            if (nums[even] % 2 == 0) {
                even += 2;
            } else if (nums[odd] % 2 == 1) {
                odd += 2;
            } else {
                // Swap misplaced even and odd
                int temp = nums[even];
                nums[even] = nums[odd];
                nums[odd] = temp;
                even += 2;
                odd += 2;
            }
        }
        return nums;
    }
}
