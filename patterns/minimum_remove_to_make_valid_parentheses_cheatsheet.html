The Minimum Remove to Make Valid Parentheses problem (LeetCode 1249) asks you to remove the minimum number of parentheses to make the input string valid (every opening parenthesis has a corresponding closing parenthesis).

ğŸ§± Pattern Template (Stack-Based Solution)

class Solution {
    public String minRemoveToMakeValid(String s) {
        Stack<Integer> stack = new Stack<>();
        Set<Integer> removeIdx = new HashSet<>();
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            if (c == '(') {
                stack.push(i);
            } else if (c == ')') {
                if (!stack.isEmpty()) {
                    stack.pop();
                } else {
                    removeIdx.add(i);
                }
            }
        }
        while (!stack.isEmpty()) {
            removeIdx.add(stack.pop());
        }
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < s.length(); i++) {
            if (!removeIdx.contains(i)) {
                sb.append(s.charAt(i));
            }
        }
        return sb.toString();
    }
}
