ğŸ“Œ What Is It?

The Longest Consecutive Sequence problem (LeetCode 128) asks you to find the length of the longest consecutive elements sequence in an unsorted array of integers. The solution must run in O(N) time.

ğŸ¯ Solution Pattern (HashSet for Sequence Start)

// Java Solution
class Solution {
    public int longestConsecutive(int[] nums) {
        Set<Integer> set = new HashSet<>();
        for (int num : nums) set.add(num);
        int maxLen = 0;
        for (int num : set) {
            if (!set.contains(num - 1)) { // sequence start
                int curr = num;
                int len = 1;
                while (set.contains(curr + 1)) {
                    curr++;
                    len++;
                }
                maxLen = Math.max(maxLen, len);
            }
        }
        return maxLen;
    }
}

ğŸ”„ Alternative Solution (Union Find)

// Java Solution (Union Find)
class Solution {
    public int longestConsecutive(int[] nums) {
        if (nums.length == 0) return 0;
        Map<Integer, Integer> map = new HashMap<>();
        UnionFind uf = new UnionFind(nums.length);
        for (int i = 0; i < nums.length; i++) {
            if (map.containsKey(nums[i])) continue; // skip duplicates
            map.put(nums[i], i);
            if (map.containsKey(nums[i] - 1)) {
                uf.union(i, map.get(nums[i] - 1));
            }
            if (map.containsKey(nums[i] + 1)) {
                uf.union(i, map.get(nums[i] + 1));
            }
        }
        return uf.maxSize();
    }
}

class UnionFind {
    int[] parent, size;
    public UnionFind(int n) {
        parent = new int[n];
        size = new int[n];
        for (int i = 0; i < n; i++) {
            parent[i] = i;
            size[i] = 1;
        }
    }
    public int find(int x) {
        if (parent[x] != x)
            parent[x] = find(parent[x]);
        return parent[x];
    }
    public void union(int x, int y) {
        int px = find(x), py = find(y);
        if (px == py) return;
        parent[py] = px;
        size[px] += size[py];
    }
    public int maxSize() {
        int max = 0;
        for (int s : size) max = Math.max(max, s);
        return max;
    }
}
