ğŸ“Œ What Is It?

The Partition Equal Subset Sum problem asks if you can partition a given array of positive integers into two subsets with equal sum.

ğŸ§± Pattern Template (Bottom-Up DP)

class Solution {
    public boolean canPartition(int[] nums) {
        int sum = 0;
        for (int num : nums) sum += num;
        if (sum % 2 != 0) return false;
        int target = sum / 2;
        boolean[] dp = new boolean[target + 1];
        dp[0] = true;
        for (int num : nums) {
            for (int i = target; i >= num; i--) {
                dp[i] = dp[i] || dp[i - num];
            }
        }
        return dp[target];
    }
}

ğŸ§± Pattern Template (Top-Down DP with Memoization)

class Solution {
    public boolean canPartition(int[] nums) {
        int sum = 0;
        for (int num : nums) sum += num;
        if (sum % 2 != 0) return false;
        int target = sum / 2;
        Boolean[][] memo = new Boolean[nums.length][target + 1];
        return dfs(nums, 0, target, memo);
    }

    private boolean dfs(int[] nums, int i, int target, Boolean[][] memo) {
        if (target == 0) return true;
        if (i == nums.length || target < 0) return false;
        if (memo[i][target] != null) return memo[i][target];
        boolean res = dfs(nums, i + 1, target, memo) || dfs(nums, i + 1, target - nums[i], memo);
        return memo[i][target] = res;
    }
}

ğŸ§± Pattern Template (2D DP Table)

class Solution {
    public boolean canPartition(int[] nums) {
        int sum = 0;
        for (int num : nums) sum += num;
        if (sum % 2 != 0) return false;
        int target = sum / 2;
        int n = nums.length;
        boolean[][] dp = new boolean[n + 1][target + 1];
        for (int i = 0; i <= n; i++) dp[i][0] = true;
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= target; j++) {
                if (j >= nums[i - 1]) {
                    dp[i][j] = dp[i - 1][j] || dp[i - 1][j - nums[i - 1]];
                } else {
                    dp[i][j] = dp[i - 1][j];
                }
            }
        }
        return dp[n][target];
    }
}
