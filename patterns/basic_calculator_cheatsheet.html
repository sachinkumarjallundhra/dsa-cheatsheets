The Basic Calculator problem (LeetCode 224, 227) asks you to evaluate a mathematical expression string containing numbers, '+', '-', '*', '/', and parentheses. You must handle operator precedence and nested parentheses.

ğŸ§± Pattern Template (Stack-Based Evaluation)

class Solution {
    public int calculate(String s) {
        Stack<Integer> stack = new Stack<>();
        int num = 0, sign = 1, result = 0;
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            if (Character.isDigit(c)) {
                num = num * 10 + (c - '0');
            } else if (c == '+') {
                result += sign * num;
                sign = 1;
                num = 0;
            } else if (c == '-') {
                result += sign * num;
                sign = -1;
                num = 0;
            } else if (c == '(') {
                stack.push(result);
                stack.push(sign);
                result = 0;
                sign = 1;
            } else if (c == ')') {
                result += sign * num;
                result *= stack.pop(); // sign before '('
                result += stack.pop(); // result before '('
                num = 0;
            }
        }
        result += sign * num;
        return result;
    }
}
