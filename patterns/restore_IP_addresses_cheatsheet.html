ğŸ“Œ What Is It?

The Restore IP Addresses problem asks you to return all possible valid IP address combinations from a given string containing only digits. Each IP address consists of four integers (0-255) separated by dots, and each integer cannot have leading zeros.

ğŸ§± Pattern Template

class Solution {
    public List<String> restoreIpAddresses(String s) {
        List<String> result = new ArrayList<>();
        backtrack(s, 0, 0, "", result);
        return result;
    }

    private void backtrack(String s, int idx, int dots, String path, List<String> result) {
        if (dots == 4 && idx == s.length()) {
            result.add(path.substring(0, path.length() - 1)); // remove last dot
            return;
        }
        if (dots > 4) return;
        for (int len = 1; len <= 3 && idx + len <= s.length(); len++) {
            String part = s.substring(idx, idx + len);
            if ((part.startsWith("0") && part.length() > 1) || Integer.parseInt(part) > 255) continue;
            backtrack(s, idx + len, dots + 1, path + part + ".", result);
        }
    }
}
